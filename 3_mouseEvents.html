Event : {{box2}}
Double Click me : {{box3}}

        const app = Vue.createApp({
            data() {
              return {
                box1: "",
                discription:"",
                box2: "",
                box3: "",
                box4X: "",
                box4Y: "",
      
              }
            },
            methods: {
              handleMouseover(e, dis) {
                  this.box1 = e.type;
                  this.discription=dis
              },
              handleMouseleave(e){
                  this.box2 = e.type
              },
              handleDoubleClick(e){
                  this.box3= e.type
              },
              handleMouseMove(e){
                  this.box4X =e.offsetX;
                  this.box4Y =e.offsetY;
      
              }
            }
          })
      
          app.mount("#app");
